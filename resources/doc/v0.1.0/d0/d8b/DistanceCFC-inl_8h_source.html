82template <typename G1, typename G2>
83void DistanceCFC<G1, G2>::initialization() {
84 // Initial guess for spherical coordinates: s2 center as view in s1
85 psi_[0] = std::atan2(this->s2_center_in_s1_[2],
86 sqrt(pow(this->s2_center_in_s1_[0], 2.0) +
87 pow(this->s2_center_in_s1_[1], 2.0)));
88 psi_[1] = std::atan2(this->s2_center_in_s1_[1], this->s2_center_in_s1_[0]);
89
90 // Initial guess for gradient/normal: along direction of s2 center as
91 // view in s1, in global coordinates
92 this->gradient_ = this->s1_->getGradientFromSpherical(
93 Eigen::Vector2d({psi_[0], psi_[1]}));
94 this->normal_ = this->gradient_.normalized();
95}

97template <typename G1, typename G2>
98void DistanceCFC<G1, G2>::results() {
99 // Optimal normal vector
100 this->res_.optimal_normal = this->normal_;
101
102 // Collision status, distance and closes points
103 this->res_.closest_point_s2 =
104 this->res_.closest_point_s1 +
105 (this->s2_center_ - this->res_.point_on_contact_space);
106 this->res_.distance =
107 (this->res_.closest_point_s2 - this->res_.closest_point_s1).norm();
108
109 if ((this->normal_).transpose() *
110 (this->res_.closest_point_s2 - this->res_.closest_point_s1) >
111 0.0) {
112 this->res_.is_collision = false;
113 } else {
114 this->res_.is_collision = true;
115 this->res_.distance = -this->res_.distance;
116 }
117
118 // Number of iterations
119 this->res_.num_iteration = this->iter_;
120}

122template <typename G1, typename G2>
123void DistanceCFC<G1, G2>::computeOptimalCFC() {
124 this->res_.closest_point_s1 =
125 this->M1_ * this->s1_->getBoundaryFromGradient(Eigen::Vector3d(
126 this->M1_.transpose() * this->gradient_)) +
127 this->s1_center_;
128 this->res_.point_on_contact_space =
129 this->cfc_obj_->getContactSpaceFromGradient(
130 Eigen::Vector3d(this->M1_.transpose() * this->gradient_)) +
131 this->s1_center_;
132}
