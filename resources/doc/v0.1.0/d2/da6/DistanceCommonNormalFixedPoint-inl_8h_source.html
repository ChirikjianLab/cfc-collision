67template <typename G1, typename G2>
68Eigen::Vector3d DistanceCommonNormalFixedPoint<G1, G2>::solve(
69 const Eigen::Vector3d& n_init, const int max_iter, const double tol,
70 int* iter, bool* flag) {
71 // Compute initial step using two normal vectors
72 Eigen::Vector3d n_01 = n_init;
73 Eigen::Vector3d r_PQ =
74 (this->M1_ * this->s2_->getBoundaryFromNormal(
75 Eigen::Vector3d(-this->M1_.transpose() * n_01)) +
76 this->s2_center_) -
77 (this->M2_ * this->s1_->getBoundaryFromNormal(
78 Eigen::Vector3d(this->M1_.transpose() * n_01)) +
79 this->s1_center_);
80 Eigen::Vector3d n_02 = (n_01 + r_PQ.normalized()).normalized();
81 Eigen::Vector3d n_new = n_02;
82
83 // Iteratively update normal vector using fixed-point iteration
84 for (*iter = 0; *iter < max_iter; ++*iter) {
85 n_new = fixedPointStep(n_01, n_02);
86
87 // Stop when difference with previous step lower than tolerance
88 if ((n_new - n_02).norm() < tol) {
89 *flag = true;
90 return n_new;
91 }
92
93 n_01 = n_02;
94 n_02 = n_new;
95 }
96
97 return n_new;
98}

100template <typename G1, typename G2>
101Eigen::Vector3d DistanceCommonNormalFixedPoint<G1, G2>::fixedPointStep(
102 const Eigen::Vector3d& n_01, const Eigen::Vector3d& n_02) {
103 Eigen::Vector3d r_MP_01 =
104 this->M1_ * this->s1_->getBoundaryFromNormal(
105 Eigen::Vector3d(this->M1_.transpose() * n_01));
106 Eigen::Vector3d r_MP_02 =
107 this->M1_ * this->s1_->getBoundaryFromNormal(
108 Eigen::Vector3d(this->M1_.transpose() * n_02));
109 Eigen::Vector3d r_SQ_01 =
110 this->M2_ * this->s2_->getBoundaryFromNormal(
111 Eigen::Vector3d(-this->M2_.transpose() * n_01));
112 Eigen::Vector3d r_SQ_02 =
113 this->M2_ * this->s2_->getBoundaryFromNormal(
114 Eigen::Vector3d(-this->M2_.transpose() * n_02));
115
116 Eigen::Matrix<double, 1, 1> mag_P_1 =
117 ((n_02 - (n_02.transpose() * n_01) * n_01).transpose() *
118 (r_MP_02 - r_MP_01)) /
119 ((n_02.cross(n_01)).transpose() * n_02.cross(n_01));
120 Eigen::Vector3d r_P_1 = -mag_P_1(0) * n_02;
121 Eigen::Matrix<double, 1, 1> mag_Q_1 =
122 ((n_02 - (n_02.transpose() * n_01) * n_01).transpose() *
123 (r_SQ_02 - r_SQ_01)) /
124 ((n_02.cross(n_01)).transpose() * n_02.cross(n_01));
125 Eigen::Vector3d r_Q_1 = -mag_Q_1(0) * n_02;
126
127 Eigen::Vector3d n_1 = ((this->s2_center_ - this->s1_center_) -
128 (r_MP_02 + r_P_1) + (r_SQ_02 + r_Q_1))
129 .normalized();
130
131 return n_1;
132}
