52template <typename G1, typename G2>
53template <typename T>
54bool CostDistanceLeastSquares<G1, G2>::operator()(const T* const psi,
55 T* cost) const {
56 // Compute normal vector
57 Eigen::Matrix<T, 2, 1> Psi = {psi[0], psi[1]};
58 Eigen::Matrix<T, 3, 1> gradient = s1_->getGradientFromSpherical(Psi);
59
60 // Compute contact space point based on normal vector
61 ClosedFormContactSpace<T, G1, G2> mink(s1_, s2_, M1_, M2_);
62 Eigen::Matrix<T, 3, 1> x_mink = mink.getContactSpaceFromGradient(gradient);
63
64 // Cost function for distance
65 Eigen::Matrix<T, 3, 1> point_distance = p0_ - x_mink;
66
67 cost[0] = point_distance(0);
68 cost[1] = point_distance(1);
69 cost[2] = point_distance(2);
70
71 return true;
72}

90template <typename G1, typename G2>
91bool DistanceLeastSquares<G1, G2>::distancePointToCFC() {
92 bool status = false;
93
94 // Retrieve spherical coordinate from initial gradient
95 this->psi_[0] = std::atan2(
96 this->gradient_[2],
97 sqrt(pow(this->gradient_[0], 2.0) + pow(this->gradient_[1], 2.0)));
98 this->psi_[1] = std::atan2(this->gradient_[1], this->gradient_[0]);
99
100 // Set up the only cost function (also known as residual). This
101 // uses auto-differentiation to obtain the derivative (jacobian)
102 setCostFunction();
103 problem_.AddResidualBlock(cost_, nullptr, this->psi_);
104
105 // Get solutions
106 ceres::Solve(options_, &problem_, &summary_);
107
108 Eigen::Vector2d Psi = {this->psi_[0], this->psi_[1]};
109 this->setGradient(this->M1_.transpose().inverse() *
110 this->s1_->getGradientFromSpherical(Psi));
111 this->normal_ = this->gradient_.normalized();
112
113 if (summary_.termination_type != ceres::NO_CONVERGENCE) {
114 status = true;
115 }
116
117 this->iter_ = summary_.num_successful_steps;
118
119 return status;
120}
