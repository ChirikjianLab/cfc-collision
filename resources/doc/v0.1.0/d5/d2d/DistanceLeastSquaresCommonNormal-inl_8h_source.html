52template <typename G1, typename G2>
53template <typename T>
54bool CostLeastSquaresCommonNormal<G1, G2>::operator()(const T* const psi,
55 T* cost) const {
56 // Compute normal vector
57 Eigen::Matrix<T, 2, 1> Psi = {psi[0], psi[1]};
58 Eigen::Matrix<T, 3, 1> gradient = s1_->getGradientFromSpherical(Psi);
59
60 // Compute contact space point based on normal vector
61 ClosedFormContactSpace<T, G1, G2> mink(s1_, s2_, M1_, M2_);
62 Eigen::Matrix<T, 3, 1> x_mink = mink.getContactSpaceFromGradient(gradient);
63
64 // Cost function for distance
65 Eigen::Matrix<T, 3, 1> gradient_point_parallel =
66 (M1_.transpose().inverse() * gradient).cross(p0_ - x_mink);
67
68 cost[0] = gradient_point_parallel(0);
69 cost[1] = gradient_point_parallel(1);
70 cost[2] = gradient_point_parallel(2);
71
72 return true;
73}
