42template <typename C>
43pair<BenchmarkMetric, vector<cfc::ContinuousDistanceResult>>
44benchmarkContinuousOptimization(
45 const vector<cfc::Shape3D>& s1,
46 const vector<pair<Eigen::Affine3d, Eigen::Affine3d>>& g1,
47 const vector<cfc::Shape3D>& s2,
48 const vector<pair<Eigen::Affine3d, Eigen::Affine3d>>& g2,
49 const cfc::ContinuousDistanceRequest& req, const int iter) {
50 // Container for benchmark results
51 BenchmarkMetric metric;
52
53 // Container of distance information
54 vector<cfc::ContinuousDistanceResult> continuous_dist_info;
55
56 for (size_t i = 0; i < size_t(iter); i++) {
57 if (i % NUM_TRIALS_DISPLAY == 0) {
58 cout << " - iteration -- " << i << endl;
59 }
60
61 // Benchmark
62 queryContinuousDistanceOptimization<C>(
63 &s1.at(i), &g1.at(i).first, &g1.at(i).second, &s2.at(i),
64 &g2.at(i).first, &g2.at(i).second, req, &metric,
65 &continuous_dist_info);
66
67 // Record results
68 metric.timeInitAvg += metric.timeInit.at(i);
69 metric.timeQueryAvg += metric.timeQuery.at(i);
70 }
71 metric.timeInitAvg /= static_cast<double>(iter);
72 metric.timeQueryAvg /= static_cast<double>(iter);
73
74 // Compute accuracy
75 computeAccuracy(continuous_dist_info, &metric);
76
77 return make_pair(metric, continuous_dist_info);
78}
