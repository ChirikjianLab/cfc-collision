70template <typename G1, typename G2>
71Eigen::Vector3d DistanceFixedPoint<G1, G2>::solve(
72 const Eigen::Vector3d& m_init) {
73 // Compute initial step using two normal vectors
74 Eigen::Vector3d m_01 = m_init;
75 Eigen::Vector3d x_mink =
76 this->cfc_obj_->getContactSpaceFromGradient(m_01) + this->s1_center_;
77 Eigen::Vector3d m_02 = this->s1_->getGradientFromDirection(Eigen::Vector3d(
78 m_01 + this->M1_.transpose() *
79 (this->s2_center_ - x_mink).normalized() * m_01.norm()));
80
81 // Iteratively update normal vector using fixed-point iteration
82 Eigen::Vector3d m_new;
83 for (this->iter_ = 0; this->iter_ < max_iter_; ++this->iter_) {
84 m_new = fixedPointStep(m_01, m_02);
85
86 // Stop when difference with previous step lower than tolerance
87 if ((m_new - m_02).norm() < tol_) {
88 flag_ = true;
89 return m_new;
90 }
91
92 m_01 = m_02;
93 m_02 = m_new;
94 }
95
96 return m_new;
97}

99template <typename G1, typename G2>
100Eigen::Vector3d DistanceFixedPoint<G1, G2>::fixedPointStep(
101 const Eigen::Vector3d& m_01, const Eigen::Vector3d& m_02) {
102 Eigen::Vector3d x_mink_1 =
103 this->M1_.transpose() *
104 this->cfc_obj_->getContactSpaceFromGradient(m_01);
105 Eigen::Vector3d x_mink_2 =
106 this->M1_.transpose() *
107 this->cfc_obj_->getContactSpaceFromGradient(m_02);
108
109 Eigen::Vector3d n_01 = m_01.normalized();
110 Eigen::Vector3d n_02 = m_02.normalized();
111
112 Eigen::Matrix<double, 1, 1> mag =
113 ((n_02 - (n_02.transpose() * n_01) * n_01).transpose() *
114 (x_mink_2 - x_mink_1)) /
115 ((n_02.cross(n_01)).transpose() * n_02.cross(n_01));
116 Eigen::Vector3d r_1 = -mag(0) * n_02;
117
118 Eigen::Vector3d m_1 = this->s1_->getGradientFromDirection(Eigen::Vector3d(
119 this->M1_.transpose() * (this->s2_center_ - this->s1_center_) -
120 (x_mink_2 + r_1)));
121
122 return m_1;
123}
