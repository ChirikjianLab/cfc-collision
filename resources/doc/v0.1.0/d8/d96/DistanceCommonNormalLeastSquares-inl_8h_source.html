55template <typename G1, typename G2>
56template <typename T>
57bool CostCommonNormalLeastSquares<G1, G2>::operator()(const T* const psi,
58 T* cost) const {
59 Eigen::Matrix<T, 2, 1> psi1 = {psi[0], psi[1]};
60 Eigen::Matrix<T, 2, 1> psi2 = {psi[2], psi[3]};
61
62 // Use common normal concept to compute \psi_2
63 Eigen::Matrix<T, 3, 1> n1_local =
64 s1_->getGradientFromSpherical(psi1).normalized();
65 Eigen::Matrix<T, 3, 1> n2_local =
66 s2_->getGradientFromSpherical(psi2).normalized();
67
68 Eigen::Matrix<T, 3, 1> x1 =
69 M1_ * s1_->getBoundaryFromNormal(n1_local) + s1_center_;
70 Eigen::Matrix<T, 3, 1> x2 =
71 M2_ * s2_->getBoundaryFromNormal(n2_local) + s2_center_;
72
73 // Cost for antiparallelism
74 Eigen::Matrix<T, 3, 1> n1 =
75 (M1_.transpose().inverse() * n1_local).normalized();
76 Eigen::Matrix<T, 3, 1> n2 =
77 (M2_.transpose().inverse() * n2_local).normalized();
78
79 Eigen::Matrix<T, 3, 1> normal_point_parallel = n1.cross(x2 - x1);
80 Eigen::Matrix<T, 3, 1> normal_normal_parallel = n1.cross(n2);
81
82 cost[0] = normal_point_parallel(0);
83 cost[1] = normal_point_parallel(1);
84 cost[2] = normal_point_parallel(2);
85 cost[3] = normal_normal_parallel(0);
86 cost[4] = normal_normal_parallel(1);
87 cost[5] = normal_normal_parallel(2);
88
89 return true;
90}

109template <typename G1, typename G2>
110bool DistanceCommonNormalLeastSquares<G1, G2>::normalOptimization() {
111 bool status = false;
112
113 // Retrieve spherical coordinate from initial gradient
114 Eigen::Vector3d n1_local = this->M1_.transpose() * this->normal_;
115 Eigen::Vector3d n2_local = -this->M2_.transpose() * this->normal_;
116
117 this->psi_[0] = std::atan2(
118 n1_local[2], sqrt(pow(n1_local[0], 2.0) + pow(n1_local[1], 2.0)));
119 this->psi_[1] = std::atan2(n1_local[1], n1_local[0]);
120 this->psi_[2] = std::atan2(
121 n2_local[2], sqrt(pow(n2_local[0], 2.0) + pow(n2_local[1], 2.0)));
122 this->psi_[3] = std::atan2(n2_local[1], n2_local[0]);
123
124 // Set up the only cost function (also known as residual). This
125 // uses auto-differentiation to obtain the derivative (jacobian)
126 cost_ =
127 new ceres::AutoDiffCostFunction<CostCommonNormalLeastSquares<G1, G2>, 6,
128 4>(
129 new CostCommonNormalLeastSquares<G1, G2>(
130 this->s1_, this->s2_, this->M1_, this->M2_, this->s1_center_,
131 this->s2_center_));
132 problem_.AddResidualBlock(cost_, nullptr, this->psi_);
133
134 // Get solutions
135 ceres::Solve(options_, &problem_, &summary_);
136
137 Eigen::Vector2d Psi = {this->psi_[0], this->psi_[1]};
138 Eigen::Vector3d gradient =
139 this->M1_ * this->s1_->getGradientFromSpherical(Psi);
140 this->setVariables(gradient.normalized());
141
142 if (summary_.termination_type != ceres::NO_CONVERGENCE) {
143 status = true;
144 }
145
146 this->res_.num_iteration = summary_.num_successful_steps;
147
148 return status;
149}
