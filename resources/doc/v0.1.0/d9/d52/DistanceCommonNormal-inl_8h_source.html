55 initialization();
56
57 // Optimize normal vectors to be anti-parallel
58 normalOptimization();

60 // Store results
61 this->res_.closest_point_s1 =
62 this->M1_ * this->s1_->getBoundaryFromNormal(Eigen::Vector3d(
63 this->M1_.transpose() * this->normal_)) +
64 this->s1_center_;
65 this->res_.closest_point_s2 =
66 this->M2_ * this->s2_->getBoundaryFromNormal(Eigen::Vector3d(
67 -this->M2_.transpose() * this->normal_)) +
68 this->s2_center_;
69 this->res_.distance =
70 (this->res_.closest_point_s2 - this->res_.closest_point_s1).norm();
71
72 if (this->normal_.transpose() *
73 (this->res_.closest_point_s2 - this->res_.closest_point_s1) >
74 0.0) {
75 this->res_.is_collision = false;
76 } else {
77 this->res_.is_collision = true;
78 this->res_.distance = -this->res_.distance;
79 }
80
81 // Necessary condition for minimum distance query: normal at closest
82 // point on s1 should be colinear with the line connecting two closest
83 // points Gradient at closest point
84 this->res_.necessary_condition =

86 this->res_.closest_point_s1))
87 .norm();
