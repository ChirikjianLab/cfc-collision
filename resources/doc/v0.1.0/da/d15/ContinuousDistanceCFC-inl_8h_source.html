135template <typename G1, typename G2>
136void ContinuousDistanceCFC<G1, G2>::initialization() {
137 // Initial guess for spherical coordinates: s2 center as view in s1
138 spherical_(0) = std::atan2(s2_center_in_s1_[2],
139 std::sqrt(std::pow(s2_center_in_s1_[0], 2.0) +
140 std::pow(s2_center_in_s1_[1], 2.0)));
141 spherical_(1) = std::atan2(s2_center_in_s1_[1], s2_center_in_s1_[0]);
142
143 // Initial guess for gradient/normal: along direction of s2 center as
144 // view in s1, in local coordinates
145 setVariables(s1_->getGradientFromSpherical(spherical_), 0.0);
146}

148template <typename G1, typename G2>
149void ContinuousDistanceCFC<G1, G2>::results() {
150 // Optimal time and normal vector
151 res_.optimal_time = t_;
152 res_.optimal_normal = g1_opt_.linear().transpose().inverse() * normal_;
153
154 // Collision status, distance and closes points
155 res_.closest_point_s2 =
156 res_.closest_point_s1 +
157 (g2_opt_.translation() - res_.point_on_contact_space);
158 res_.distance = (res_.closest_point_s2 - res_.closest_point_s1).norm();
159
160 if (res_.optimal_normal.transpose() *
161 (res_.closest_point_s2 - res_.closest_point_s1) >
162 0.0 &&
163 res_.distance > 1e-3) {
164 res_.is_collision = false;
165 } else {
166 res_.is_collision = true;
167 res_.distance = -res_.distance;
168 }
169
170 // Number of iterations
171 res_.num_iteration = iter_;
172}
