41template <typename P>
42EMesh getMesh(P points) {
43 EMesh mesh;
44
45 // Convert vertices to list of fcl::Vector3
46 for (size_t i = 0; i < points.x.size(); ++i) {
47 fcl::Vector3<double> v;
48 v << points.x.at(i), points.y.at(i), points.z.at(i);
49 mesh.vertices.push_back(v);
50 }
51
52 // Compute mesh faces from parametric surface
53 Eigen::MatrixXd mesh_surf = getMeshFaceFromSurf(points);
54
55 // Convert mesh faces to list of fcl::Triangles
56 for (size_t i = 0; i < mesh_surf.rows(); ++i) {
57 fcl::Triangle triangle(mesh_surf(i, 0), mesh_surf(i, 1),
58 mesh_surf(i, 2));
59 mesh.triangles.push_back(triangle);
60 }
61
62 return mesh;
63}

65template <typename G>
66cfc::ParametricPoints getBoundary3D(const G* obj, const int num) {
67 double eta;
68 double omega;
69 cfc::ParametricPoints X;
70 Eigen::Vector3d x;
71
72 for (int i = 0; i < num; ++i) {
73 eta = -pi / 2.0 + i * pi / (num - 1);
74
75 for (int j = 0; j < num; ++j) {
76 omega = -pi + j * (2.0 * pi) / (num - 1);
77
78 x = obj->getBoundaryFromSpherical(Eigen::Vector2d(eta, omega));
79
80 // Transform object to world frame
81 const Eigen::Quaterniond quat = obj->quaternion();
82 const Eigen::Vector3d center(
83 {obj->pos()[0], obj->pos()[1], obj->pos()[2]});
84 x = quat.matrix() * x + center;
85
86 X.x.push_back(x(0));
87 X.y.push_back(x(1));
88 X.z.push_back(x(2));
89 }
90 }
91 return X;
92}

111template <typename P>
112Eigen::MatrixXd getMeshFaceFromSurf(P points) {
113 auto n = static_cast<int>(std::sqrt(points.x.size()));
114 auto Num = (n - 1) * (n - 1);
115
116 Eigen::ArrayXd q((n - 1) * (n - 1));
117 for (auto i = 0; i < n - 1; ++i) {
118 q.segment(i * (n - 1), (n - 1)) =
119 Eigen::ArrayXd::LinSpaced(n - 1, i * n, (i + 1) * n - 2);
120 }
121
122 Eigen::MatrixXd faces = Eigen::MatrixXd::Zero(2 * Num, 3);
123 faces.block(0, 0, Num, 1) = q + n + 1;
124 faces.block(0, 1, Num, 1) = q + n;
125 faces.block(0, 2, Num, 1) = q;
126 faces.block(Num, 0, Num, 1) = q;
127 faces.block(Num, 1, Num, 1) = q + 1;
128 faces.block(Num, 2, Num, 1) = q + n + 1;
129
130 return faces;
131}
