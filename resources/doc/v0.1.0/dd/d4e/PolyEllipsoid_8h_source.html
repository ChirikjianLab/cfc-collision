252 template <typename T>
253 Eigen::Matrix<T, 3, 3> getShapeParamOctant(
254 const Eigen::Matrix<T, 3, 1> point) const {
255 Eigen::Matrix<T, 3, 3> A;
256 A.setIdentity();
257
258 A(0, 0) = point[0] > 0.0 ? T(shape_.a[0]) : T(shape_.b[0]);
259 A(1, 1) = point[1] > 0.0 ? T(shape_.a[1]) : T(shape_.b[1]);
260 A(2, 2) = point[2] > 0.0 ? T(shape_.a[2]) : T(shape_.b[2]);
261
262 return A;
